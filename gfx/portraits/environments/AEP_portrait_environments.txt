	cubemap = "gfx/portraits/environments/frontend_west_01.dds"
	cubemap_intensity = 0.2

	lights = {

		#								KEY LIGHT								#

			falloff = 10

			# For spot lights

			inner_cone_angle = 0

		#########################################################################

			# For spot lights and directional lights

			# For spot lights and point lights
			radius = 1000
			falloff = 10

			# For spot lights
			outer_cone_angle = 55
			inner_cone_angle = 0
		}
	}

		look_at = { 0 0 0 }

	}
